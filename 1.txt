sudo tee /etc/nginx/sites-available/minio <<EOF
server {
    listen 80;
    server_name s3.bizkit.cloud;

    # Redirect all HTTP -> HTTPS
    return 301 https://\$host\$request_uri;
}

server {
    listen 443 ssl http2;
    server_name s3.bizkit.cloud;

    # TLS certs (using Let's Encrypt or your own)
    ssl_certificate     /etc/letsencrypt/live/shrt.wiki/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/shrt.wiki/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;

    # Console (web UI)
    location / {
        proxy_pass http://127.0.0.1:9001;
        proxy_set_header Host             \$host;
        proxy_set_header X-Real-IP        \$remote_addr;
        proxy_set_header X-Forwarded-For  \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
    }

    # S3 API endpoints (optional)
    location /s3/ {
        rewrite ^/s3/(.*)\$ /\$1 break;
        proxy_pass http://127.0.0.1:9000;
        proxy_set_header Host             \$host;
        proxy_set_header X-Real-IP        \$remote_addr;
        proxy_set_header X-Forwarded-For  \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
    }
}
EOF
